### FX Facade

- [ ] **Step 12: Create `js/fx.js`**
```js
// js/fx.js — FORGE unified FX facade
// Delegates to fx-sound.js, fx-haptic.js, fx-visuals.js
// Usage: fx.sound('sndPR'); fx.haptic('hapPR'); fx.burst('PR', x, y);
(function() {
  window.fx = {
    sound(name, ...args) {
      try { window[name]?.(...args); } catch(e) {}
    },
    haptic(name, ...args) {
      try { window[name]?.(...args); } catch(e) {}
    },
    burst(type, x, y) {
      try { window[`burst${type}`]?.(x, y); } catch(e) {}
    },
    flash(type) {
      try { window[`flash${type}`]?.(); } catch(e) {}
    }
  };
})();
```

- [ ] **Step 4: Add wizard logic to `js/onboarding-controls.js`**
```js
// Onboarding wizard — shows once for new users (no name + no workouts)
(function initOnboarding() {
  const shown = localStorage.getItem('forge_onboarding_done');
  const hasName = localStorage.getItem('forge_username');
  const hasWorkouts = localStorage.getItem('forge_workouts');
  if (shown || hasName || hasWorkouts) return;

  document.getElementById('onboarding-overlay').style.display = 'flex';

  window._obNext = function(step) {
    fx.sound('sndTap'); fx.haptic('hapTap');
    if (step === 2) {
      const name = document.getElementById('ob-name').value.trim();
      if (!name) return;
      localStorage.setItem('forge_username', name);
    }
    document.getElementById(`ob-step-${step}`).hidden = true;
    document.getElementById(`ob-step-${step + 1}`).hidden = false;
  };

  window._obGoal = function(goal) {
    localStorage.setItem('forge_goal', goal);
    fx.sound('sndTap');
    document.getElementById('ob-step-3').hidden = true;
    document.getElementById('ob-step-4').hidden = false;
  };

  window._obFinish = function(level) {
    localStorage.setItem('forge_experience', level);
    localStorage.setItem('forge_onboarding_done', '1');
    fx.sound('sndSessionStart'); fx.haptic('hapSave');
    document.getElementById('onboarding-overlay').style.display = 'none';
  };
})();
```

function animatedDraw(ctx, points, color, duration=800) {
  const start = Date.now();
  const tick = () => {
    const pct = Math.min((Date.now() - start) / duration, 1);
    const count = Math.floor(points.length * pct);
    ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
    ctx.beginPath();
    points.slice(0, count).forEach((p, i) => i === 0 ? ctx.moveTo(p.x, p.y) : ctx.lineTo(p.x, p.y));
    ctx.strokeStyle = color; ctx.lineWidth = 2; ctx.stroke();
    if (pct < 1) requestAnimationFrame(tick);
  };
  tick();
}
